\section{Question No-6.20}
An unbiased dice is thrown twice. Let the event A be 'odd number on the first throw' and B be 'Odd number on second throw'. Check the independence of event A and B. 
\section{Solution}
\begin{lemma}
If A and B are independent events then the property can be expressed as \label{L1}
\begin{align}
  \pr{A|B}=\pr{A}.   
\end{align} 
\end{lemma} 
\begin{enumerate}
\item According to question,the events are:-
\begin{table}[ht!]
\begin{tabular}{|l|l|}
\hline
\textbf{Events} & \textbf{Description}                         \\ \hline
$A$ & Odd number on first throw
\\ \hline
$B$ & Odd number on second throw 
\\ \hline
$AB$ & Odd Numbers appears on both throws \\ \hline
\end{tabular}
\end{table}
\item Let $X_0$ and $X_1$ be the random variables representing the numbers we get  when a dice is thrown for first and second time respectively.
\begin{align}
    X_0\epsilon\{1,2,3,4,5,6\}
    \\
     X_1\epsilon\{1,2,3,4,5,6\}
\end{align}
\item Also, the probability
\begin{align}
\pr{X=i}=\begin{cases}
	\frac{1}{6}& 1 \le i \le 6\\
	0 & otherwise
	\end{cases}
\end{align}
\item From the above information we have,
\begin{align}
 \pr{A}&=\sum_{i=1,3,5} \pr{X_0=i}= \frac{1}{2}
 \\
 \pr{B}&=\sum_{i=1,3,5} \pr{X_1=i}= \frac{1}{2}
\\
    \pr{AB} &= \frac{1}{4}
\end{align}
\item Now to check whether the events are independent we use Lemma \ref{L1}
\begin{align}
   \pr{A|B} &= \frac{\pr{AB}}{\pr{B}} \\
   &= \frac{1}{2} \\ 
   &= \pr{A}
\end{align}
\item Thus $\pr{A|B}=\pr{A}$ which implies the  events are \textbf{independent}.
\end{enumerate}
